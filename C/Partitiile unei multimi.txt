Partitiile unei multimi
#include<stdio.h>
#include<conio.h>

int n, x[10], max[10];

void afis (int p) //p este nr de submultimi din partitia curenta
{ for( int i=1;i<p;i++)
{ printf("{ ");
  for(int j=1;j<=n;j++)
	  if(x[j]==x[i]) printf("%d ",j);
  printf("}+");
  
}
printf("{ ");
for(int j=1;j<=n;j++)
  if(x[j]==p) printf("%d ",j);
printf("}");
putchar('\n');
}

void partitii(int i)
{ for (int j=1;j<=1+max[i-1];j++)
{ x[i]=j;
  if(x[i]>max[i-1]) max[i]=x[i];
  else max[i]=max[i-1];
  if(i==n) afis(max[i]);
  else partitii(i+1);
}
}

void main(void)
{ printf("n=...");
scanf("%d",&n);
max[0]=0;
partitii(1);
getch();
}
